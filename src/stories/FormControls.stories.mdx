## Inset Inputs

Try not to mix Inset style inputs with regular inputs in the same form.

<Canvas>
  <Story name="Inset Inputs">
    {() => {
      const [formValues, setFormValues] = useState({
        emailInput: '',
        passwordInput: '',
        notesInput: '',
        toggleValue: false,
      });
      const options = [
        {
          value: 'chocolate',
          label: 'Chocolate',
        },
        {
          value: 'strawberry',
          label: 'Strawberry',
        },
        {
          value: 'vanilla',
          label: 'Vanilla',
        },
      ];
      const handleChange = (key, value) => {
        setFormValues(prevFields => ({
          ...prevFields,
          [key]: value,
        }));
      };
      return (
        <Box gap="md">
          <TextInputInset
            label="Email"
            id="emailInput"
            value={formValues.textInputMd}
            onChange={event => handleChange('emailInput', event.target.value)}
          />
          <TextInputInset
            label="Password"
            type="password"
            id="passwordInput"
            value={formValues.textInputMd}
            onChange={event =>
              handleChange('passwordInput', event.target.value)
            }
          />
          <TextareaInputInset
            label="Notes"
            id="notesInput"
            value={formValues.textInputMd}
            onChange={event => handleChange('notesInput', event.target.value)}
          />
          <SelectInputInset
            id="floatingSelectNative"
            label="Flavor"
            onChange={event => handleChange('selectInput', event.target.value)}
            options={options}
            value={formValues.selectInput}
            size="md"
          />
        </Box>
      );
    }}
  </Story>
</Canvas>
